\item 2. Compute $13^{2019}$ (mod 37) 

\begin{itemize}

\item Answer: $a^{p-1} $ = 1 mod p 
\item $a^{p-1} $ mod p = 1
\item $13^{37-1}$  mod 37 = 1
\item $13^{36}$  mod 37 = 1
\item 2019 = 36*56+3
\item $13^{2019}$  mod 37 = $(13^{36}  mod 37)^{56}$ mod 37 * $13^{3}$ mod 37
\item = (1 mod 37) * ($13^{3}$ mod 37)
\item = ($13^{3}$ mod 37)
\item = 2197 mod 37 
\item = 14 

\end{itemize} 

\end{itemize}
